CREATE TABLE `report_found` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `name` VARCHAR(255) NOT NULL,
  `item_name` VARCHAR(255) NOT NULL,
  `description` TEXT DEFAULT NULL,
  `location` VARCHAR(255) DEFAULT NULL,
  `contact` VARCHAR(50) NOT NULL,
  `email` VARCHAR(255) NOT NULL,
  `date` DATE DEFAULT NULL,
  `photo` VARCHAR(255) DEFAULT NULL,
  `report_date` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `report_lost` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `name` VARCHAR(255) NOT NULL,
  `item_name` VARCHAR(255) NOT NULL,
  `description` TEXT DEFAULT NULL,
  `location` VARCHAR(255) DEFAULT NULL,
  `contact` VARCHAR(50) NOT NULL,
  `email` VARCHAR(255) NOT NULL,
  `date` DATE DEFAULT NULL,
  `photo` VARCHAR(255) DEFAULT NULL,
  `report_date` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `claimed_items` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `item_name` VARCHAR(255) DEFAULT NULL,
  `description` TEXT DEFAULT NULL,
  `location` VARCHAR(255) DEFAULT NULL,
  `photo` VARCHAR(255) DEFAULT NULL,
  `finder_name` VARCHAR(255) DEFAULT NULL,
  `finder_contact` VARCHAR(50) DEFAULT NULL,
  `finder_email` VARCHAR(255) DEFAULT NULL,
  `claimer_name` VARCHAR(255) DEFAULT NULL,
  `claimer_contact` VARCHAR(50) DEFAULT NULL,
  `claimer_email` VARCHAR(255) DEFAULT NULL,
  `claim_date` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
